import pandas as pd
import gensim
from gensim.models import Word2Vec
import torch
import torch.nn as nn
import torch.optim as optim
import numpy as np
from sklearn.preprocessing import LabelEncoder
from sklearn.model_selection import train_test_split
from sklearn.metrics import f1_score, recall_score
from torch.utils.data import DataLoader, TensorDataset
from tqdm import tqdm

device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

# 加载数据
def load_data():
    train_data = pd.read_csv('datasets/train_1_4.csv')
    test_data = pd.read_csv('datasets/test_1_4.csv')
    return train_data, test_data

# 文本预处理：分词
def preprocess_data(train_data, test_data):
    train_sentences = [sentence.split() for sentence in train_data['processed']]
    test_sentences = [sentence.split() for sentence in test_data['processed']]
    return train_sentences, test_sentences

# 训练 CBOW / Skip-gram 词向量模型
def train_word2vec_model(sentences):
    model_cbow = Word2Vec(sentences, vector_size=128, window=2, sg=0, min_count=1)
    # 为 <unk> 词预分配空间
    model_cbow.wv.add_vectors(['<unk>'], np.random.normal(size=(1, model_cbow.vector_size)))
    model_cbow.save("model/cbow_model_1_4.model")
    return model_cbow

# 将句子转换为词向量平均值
def sentence_to_vector(sentence, model, unk_token='<unk>'):
    words = sentence.split()
    word_vectors = []

    for word in words:
        if word in model.wv:
            word_vectors.append(model.wv[word])
        else:
            if unk_token in model.wv:
                word_vectors.append(model.wv[unk_token])

    if not word_vectors:
        # 句子完全为空或无词向量，用零向量代替
        return np.zeros(model.vector_size)

    return np.mean(word_vectors, axis=0)


# 定义情感分类模型
class SentimentModel(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim):
        super(SentimentModel, self).__init__()
        self.fc1 = nn.Linear(input_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, output_dim)
        self.relu = nn.ReLU()
        self.softmax = nn.Softmax(dim=1)

    def forward(self, x):
        x = self.relu(self.fc1(x))
        x = self.fc2(x)
        return self.softmax(x)

# 训练情感分类模型
def train_sentiment_model(train_data, test_data, model_cbow, unk_token='<unk>'):
    # 训练集数据处理
    X_train = np.array([sentence_to_vector(sentence, model_cbow, unk_token) for sentence in train_data['processed']])
    y_train = LabelEncoder().fit_transform(train_data['target'])

    # 测试集数据处理
    X_test = np.array([sentence_to_vector(sentence, model_cbow, unk_token) for sentence in test_data['processed']])
    y_test = LabelEncoder().fit_transform(test_data['target'])

    # 创建DataLoader
    train_tensor = TensorDataset(torch.tensor(X_train, dtype=torch.float32),
                                 torch.tensor(y_train, dtype=torch.long))
    test_tensor = TensorDataset(torch.tensor(X_test, dtype=torch.float32),
                                torch.tensor(y_test, dtype=torch.long))

    train_loader = DataLoader(train_tensor, batch_size=256, shuffle=True)
    test_loader = DataLoader(test_tensor, batch_size=256)

    # 模型初始化
    model = SentimentModel(input_dim=128, hidden_dim=128, output_dim=2).to(device)
    criterion = nn.CrossEntropyLoss()
    optimizer = optim.Adam(model.parameters(), lr=0.001)

    best_acc = 0.0
    progress_bar = tqdm(range(30), desc="Total Progress", position=0, colour='GREEN')

    for epoch in progress_bar:
        # 训练阶段
        model.train()
        epoch_loss = 0
        train_batch_bar = tqdm(train_loader, desc=f"Train Epoch {epoch + 1}", leave=False, colour='BLUE')

        for inputs, labels in train_batch_bar:
            inputs, labels = inputs.to(device), labels.to(device)

            optimizer.zero_grad()
            outputs = model(inputs)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()

            epoch_loss += loss.item()
            train_batch_bar.set_postfix(loss=f"{loss.item():.4f}")

        # 测试阶段
        model.eval()
        correct, total = 0, 0
        test_batch_bar = tqdm(test_loader, desc=f"Testing Epoch {epoch + 1}", leave=False, colour='MAGENTA')

        with torch.no_grad():
            for inputs, labels in test_batch_bar:
                inputs, labels = inputs.to(device), labels.to(device)
                outputs = model(inputs)
                _, predicted = torch.max(outputs, 1)

                total += labels.size(0)
                correct += (predicted == labels).sum().item()
                test_batch_bar.set_postfix(acc=f"{correct / total:.2%}")

        epoch_acc = correct / total
        progress_bar.set_postfix(best_acc=f"{best_acc:.2%}", current_acc=f"{epoch_acc:.2%}")

        # 保存最佳模型
        if epoch_acc > best_acc:
            best_acc = epoch_acc
            torch.save(model.state_dict(), "sentiment_model/best_cbow_model_1_4.pth")

    print(f"\nBest Test Accuracy: {best_acc:.2%}")
    return model


# 修改后的主函数
def main():
    # 加载数据
    train_data, test_data = load_data()

    # 预处理（需包含之前的<unk>处理）
    train_sentences, test_sentences = preprocess_data(train_data, test_data)

    # 训练词向量
    model_cbow = train_word2vec_model(train_sentences)

    # 训练并测试情感模型
    sentiment_model = train_sentiment_model(train_data, test_data, model_cbow)


if __name__ == "__main__":
    main()
